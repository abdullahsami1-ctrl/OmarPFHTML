Usage

During Many times people get busy and do not have time to feed ther pets, thereby often leaving their pets hungry

That is why i have created this Automated pet feeder.

Working

When a pet comes close, the Ultrasonic Sensor senses the pet by sending out Ultrasonic Waves and when reflected at the required distance sends the current to the Arduino UNO to activate the Servo Motor thereby opening the flap and dropping the food, which is then caught by the container ready for the pet to eat.

Code


#include


Servo feederServo;


const int trigPin = 9;

const int echoPin = 8;


long duration;

int distance;


// ---- Reliable ultrasonic reading ----

int getDistance() {

digitalWrite(trigPin, LOW);

delayMicroseconds(2);

digitalWrite(trigPin, HIGH);

delayMicroseconds(10);

digitalWrite(trigPin, LOW);


duration = pulseIn(echoPin, HIGH, 30000); // timeout 30ms

if (duration == 0) return 999; // no reading = ignore


return duration * 0.034 / 2;

}


void setup() {

feederServo.attach(5); // <<< FIXED: on pin 5 (no timer conflict)

feederServo.write(10); // safe start angle


pinMode(trigPin, OUTPUT);

pinMode(echoPin, INPUT);


Serial.begin(9600);

}


void loop() {

distance = getDistance();

Serial.println(distance);


// Noise filtering

if (distance < 5 || distance > 300) return;


// Real trigger

if (distance < 20) {

// open

for (int pos = 10; pos <= 80; pos++) {

feederServo.write(pos);

delay(10);

}


delay(1000);


// close

for (int pos = 80; pos >= 10; pos--) {

feederServo.write(pos);

delay(10);

}


delay(5000);

}


delay(200);

}
